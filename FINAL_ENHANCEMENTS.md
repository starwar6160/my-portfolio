```javascript
// 页面加载时检查保存的密码
window.addEventListener('DOMContentLoaded', () => {
  const savedPassword = localStorage.getItem('analyticsPassword');
  if (savedPassword) {
    password = savedPassword;
    loadData(); // 自动加载数据
  }
});

// 登录时保存密码
function login() {
  password = document.getElementById('passwordInput').value;
  localStorage.setItem('analyticsPassword', password); // 保存
  loadData();
}

// 登出时清除密码
function logout() {
  localStorage.removeItem('analyticsPassword'); // 清除
  // ... 其他登出逻辑
}

// 密码错误时清除
if (response.status === 401) {
  localStorage.removeItem('analyticsPassword'); // 清除无效密码
  showError();
}
```
